Table User {
  userId integer [primary key]
  email varchar [unique, not null]
  password varchar [not null]
}

Table Orchard {
  orchardId integer [primary key]
  name varchar
  frequencyCheckDays integer // this is a temporary measure - we will replace this with other details of the vineyard
}

Table OrchardUser {
  userId integer [primary key]
  orchardId integer [primary key]
}

Table Record {
  recordId integer [primary key]
  orchardId integer
}

Table Disease {
  diseaseId integer [primary key]
  diseaseName varchar
}

Table RecordDisease {
  recordId integer [primary key]
  diseaseId integer [primary key]
}

// Relationships
Ref: OrchardUser.userId > User.userId
Ref: OrchardUser.orchardId > Orchard.orchardId
Ref: Record.orchardId > Orchard.orchardId
Ref: RecordDisease.recordId > Record.recordId
Ref: RecordDisease.diseaseId > Disease.diseaseId
